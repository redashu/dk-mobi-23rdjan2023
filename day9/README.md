### final YAML of mysql Deployment with configmap & secret 

```
apiVersion: v1
data:
  MYSQL_DATABASE: MobiEmp
  MYSQL_USER: ashu 
kind: ConfigMap
metadata:
  creationTimestamp: null
  name: ashu-db-details
---
apiVersion: v1
data:
  rootpass: TW9iaVBhc3NAMDk=
kind: Secret
metadata:
  creationTimestamp: null
  name: ashu-db-cred
---
apiVersion: apps/v1
kind: Deployment
metadata:
  creationTimestamp: null
  labels:
    app: ashu-db-server
  name: ashu-db-server
spec:
  replicas: 1
  selector:
    matchLabels:
      app: ashu-db-server
  strategy: {}
  template:
    metadata:
      creationTimestamp: null
      labels:
        app: ashu-db-server
    spec:
      containers:
      - image: mysql:5.6
        name: mysql
        ports:
        - containerPort: 3306
        resources: {}
        envFrom:
        - configMapRef: # keyword to call CM 
            name: ashu-db-details # name of configmap 
        env: # calling values 
        - name: MYSQL_ROOT_PASSWORD # this variable want a value 
          valueFrom: 
            secretKeyRef: # keyword to call secret 
              name: ashu-db-cred # name of secret 
              key: rootpass  # key of secret 
status: {}
```
