### dockerfile 

```
FROM oraclelinux:8.4
ENV code=blue
RUN mkdir /test
COPY hello.sh /test/
WORKDIR /test
RUN chmod +x hello.sh
ENTRYPOINT ["./hello.sh"]
```

### hello.sh 

```
#!/bin/bash

if [ "$code" == "blue"  ]
then
    while [ true ]
    do  
        echo "Hello i am code Blue !!!"
        sleep 10 
    done
elif [ "$code" == "black"  ]
then
    while [ true ]
    do
        echo "Hello i am code black !!!"
        sleep 10
    done
fi 
```

### lets add ENV section in Deployment file 

```
apiVersion: apps/v1
kind: Deployment
metadata:
  creationTimestamp: null
  labels:
    app: ashu-test
  name: ashu-test
spec:
  replicas: 1
  selector:
    matchLabels:
      app: ashu-test
  strategy: {}
  template:
    metadata:
      creationTimestamp: null
      labels:
        app: ashu-test
    spec:
      containers:
      - image: docker.io/dockerashu/testing:v1
        name: testing
        resources: {}
        env: # to create / use ENV variable of docker images 
        - name: code # name of ENV 
          value: black # value of ENV 
status: {}

```
